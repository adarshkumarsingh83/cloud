#logout from PWS 
$ cf logout

#Push the app to PWS:
$ cf push 

# View a snapshot of recent logs:
$ cf logs <APP_NAME> --recent

#stream live logs:
$ cf logs <APP_NAME>

#to delete the app from PWS:
$ cf delete <APP_NAME> [-r] [-f]
-f
Force deletion without confirmation
-r
Also delete any mapped routes
